import swisseph as swe
import datetime

# Set path to Swiss Ephemeris files
swe.set_ephe_path("C:/swisseph-master/")  # Adjust path if needed

# Function to calculate planetary positions for a given birthdate
def get_birth_chart(year, month, day, hour=12.0):
    jd = swe.julday(year, month, day, hour)
    planets = {
        "Sun": swe.SUN, "Moon": swe.MOON, "Mercury": swe.MERCURY,
        "Venus": swe.VENUS, "Mars": swe.MARS, "Jupiter": swe.JUPITER,
        "Saturn": swe.SATURN, "Uranus": swe.URANUS, "Neptune": swe.NEPTUNE,
        "Pluto": swe.PLUTO, "North Node": swe.MEAN_NODE
    }
    
    chart = {}
    for name, planet in planets.items():
        pos, _, _, _ = swe.calc_ut(jd, planet)
        chart[name] = f"{pos:.2f}Â°"
    
    return chart

# Example: Get birth chart for March 15, 1990 at 12:00 PM
birth_chart = get_birth_chart(1990, 3, 15, 12.0)

# Display the results
for planet, position in birth_chart.items():
    print(f"{planet}: {position}")
